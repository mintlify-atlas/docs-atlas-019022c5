### Stored Procedures

#### 1. `proc_validate_sensor_readings(batch_id UUID)`

```sql
CREATE OR REPLACE FUNCTION staging.proc_validate_sensor_readings(p_batch_id UUID)
RETURNS JSON AS $$
DECLARE
    v_total INT;
    v_valid INT := 0;
    v_invalid INT := 0;
    v_errors TEXT[];
BEGIN
    -- Count total records in batch
    SELECT COUNT(*) INTO v_total 
    FROM staging.sensor_readings_raw 
    WHERE batch_id = p_batch_id AND validation_status = 'pending';
    
    -- Validate each record
    WITH validated_records AS (
        SELECT 
            r.id,
            r.time,
            r.sensor_id,
            r.greenhouse_id,
            r.tenant_id,
            r.sensor_type,
            r.value,
            r.unit,
            r.metadata,
            CASE
                -- Check greenhouse_id is valid UUID
                WHEN r.greenhouse_id IS NULL THEN ARRAY['greenhouse_id is NULL']
                -- Check tenant_id is valid UUID
                WHEN r.tenant_id IS NULL THEN ARRAY['tenant_id is NULL']
                -- Check value is within range for sensor type
                WHEN EXISTS (
                    SELECT 1 FROM staging.validation_rules vr
                    WHERE vr.sensor_type = r.sensor_type
                      AND vr.is_active = TRUE
                      AND (r.value < vr.min_value OR r.value > vr.max_value)
                ) THEN ARRAY['Value ' || r.value || ' out of range for ' || r.sensor_type]
                -- Check required fields
                WHEN r.time IS NULL THEN ARRAY['time is NULL']
                WHEN r.sensor_id IS NULL THEN ARRAY['sensor_id is NULL']
                WHEN r.value IS NULL THEN ARRAY['value is NULL']
                ELSE NULL
            END AS errors
        FROM staging.sensor_readings_raw r
        WHERE r.batch_id = p_batch_id AND r.validation_status = 'pending'
    )
    -- Insert valid records into validated table
    INSERT INTO staging.sensor_readings_validated (
        raw_reading_id, batch_id, time, sensor_id, greenhouse_id, 
        tenant_id, sensor_type, value, unit, metadata
    )
    SELECT 
        id, p_batch_id, time, sensor_id, greenhouse_id,
        tenant_id, sensor_type, value, unit, metadata
    FROM validated_records
    WHERE errors IS NULL;
    
    GET DIAGNOSTICS v_valid = ROW_COUNT;
    
    -- Update invalid records
    UPDATE staging.sensor_readings_raw r
    SET 
        validation_status = 'invalid',
        validation_errors = v.errors
    FROM validated_records v
    WHERE r.id = v.id AND v.errors IS NOT NULL;
    
    GET DIAGNOSTICS v_invalid = ROW_COUNT;
    
    -- Update bulk import log
    UPDATE staging.bulk_import_log
    SET 
        completed_at = NOW(),
        duration_seconds = EXTRACT(EPOCH FROM (NOW() - started_at)),
        status = CASE WHEN v_invalid = 0 THEN 'completed' ELSE 'partially_completed' END,
        total_records = v_total,
        successful_records = v_valid,
        failed_records = v_invalid
    WHERE batch_id = p_batch_id;
    
    RETURN json_build_object(
        'total', v_total,
        'valid', v_valid,
        'invalid', v_invalid
    );
END;
$$ LANGUAGE plpgsql;
```

#### 2. `proc_migrate_staging_to_production(batch_id UUID, delete_after BOOLEAN)`

```sql
CREATE OR REPLACE FUNCTION staging.proc_migrate_staging_to_production(
    p_batch_id UUID,
    p_delete_after BOOLEAN DEFAULT FALSE
)
RETURNS JSON AS $$
DECLARE
    v_migrated INT;
BEGIN
    -- Bulk insert validated data into production
    INSERT INTO iot.sensor_readings (
        time, sensor_id, greenhouse_id, tenant_id, 
        sensor_type, value, unit, metadata
    )
    SELECT 
        time, sensor_id, greenhouse_id, tenant_id,
        sensor_type, value, unit, metadata
    FROM staging.sensor_readings_validated
    WHERE batch_id = p_batch_id;
    
    GET DIAGNOSTICS v_migrated = ROW_COUNT;
    
    -- Update validation status
    UPDATE staging.sensor_readings_validated
    SET validation_status = 'migrated'
    WHERE batch_id = p_batch_id;
    
    -- Optional cleanup
    IF p_delete_after THEN
        DELETE FROM staging.sensor_readings_validated WHERE batch_id = p_batch_id;
        DELETE FROM staging.sensor_readings_raw WHERE batch_id = p_batch_id;
    END IF;
    
    -- Update bulk import log
    INSERT INTO staging.bulk_import_log (batch_id, operation_type, status, successful_records)
    VALUES (p_batch_id, 'migration', 'completed', v_migrated);
    
    RETURN json_build_object(
        'migrated', v_migrated,
        'status', 'completed'
    );
END;
$$ LANGUAGE plpgsql;
```

#### 3. `proc_cleanup_staging(days_to_keep INT)`

```sql
CREATE OR REPLACE FUNCTION staging.proc_cleanup_staging(p_days_to_keep INT DEFAULT 30)
RETURNS JSON AS $$
DECLARE
    v_deleted_raw INT;
    v_deleted_validated INT;
BEGIN
    -- Delete old raw records
    DELETE FROM staging.sensor_readings_raw
    WHERE import_timestamp < NOW() - (p_days_to_keep || ' days')::INTERVAL;
    GET DIAGNOSTICS v_deleted_raw = ROW_COUNT;
    
    -- Delete old validated records
    DELETE FROM staging.sensor_readings_validated
    WHERE validated_at < NOW() - (p_days_to_keep || ' days')::INTERVAL;
    GET DIAGNOSTICS v_deleted_validated = ROW_COUNT;
    
    RETURN json_build_object(
        'deleted_raw', v_deleted_raw,
        'deleted_validated', v_deleted_validated
    );
END;
$$ LANGUAGE plpgsql;
```

### Continuous Aggregates

#### `iot.sensor_readings_hourly`

```sql
CREATE MATERIALIZED VIEW iot.sensor_readings_hourly
WITH (timescaledb.continuous) AS
SELECT
    time_bucket('1 hour', time) AS time,
    greenhouse_id,
    tenant_id,
    sensor_type,
    AVG(value) AS avg_value,
    MIN(value) AS min_value,
    MAX(value) AS max_value,
    STDDEV(value) AS stddev_value,
    PERCENTILE_CONT(0.50) WITHIN GROUP (ORDER BY value) AS percentile_50,
    PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY value) AS percentile_95,
    COUNT(*) AS count_readings
FROM iot.sensor_readings
GROUP BY time_bucket('1 hour', time), greenhouse_id, tenant_id, sensor_type;

-- Refresh policy (every 1 hour)
SELECT add_continuous_aggregate_policy('iot.sensor_readings_hourly',
    start_offset => INTERVAL '3 hours',
    end_offset => INTERVAL '1 hour',
    schedule_interval => INTERVAL '1 hour');
```

#### `iot.sensor_readings_daily_by_tenant`

```sql
CREATE MATERIALIZED VIEW iot.sensor_readings_daily_by_tenant
WITH (timescaledb.continuous) AS
SELECT
    time_bucket('1 day', time) AS day,
    tenant_id,
    COUNT(*) AS total_readings,
    COUNT(DISTINCT greenhouse_id) AS unique_greenhouses,
    COUNT(DISTINCT sensor_id) AS unique_sensors,
    AVG(CASE WHEN sensor_type = 'TEMPERATURE' THEN value END) AS avg_temperature,
    AVG(CASE WHEN sensor_type = 'HUMIDITY' THEN value END) AS avg_humidity
FROM iot.sensor_readings
GROUP BY time_bucket('1 day', time), tenant_id;

-- Refresh policy (every 6 hours)
SELECT add_continuous_aggregate_policy('iot.sensor_readings_daily_by_tenant',
    start_offset => INTERVAL '1 week',
    end_offset => INTERVAL '1 day',
    schedule_interval => INTERVAL '6 hours');
```

### V22: Migrate UUID to BIGINT (780 lines)

**Date:** 2026-01-08

**Purpose:** Migrate all UUID primary keys to BIGINT for 40% performance improvement.

**Impact:**
- Index size reduced by 50%
- JOIN queries 40% faster
- Auto-increment IDs instead of UUID generation overhead

**Tables Migrated:**
1. tenants (root)
2. users
3. greenhouses
4. sectors
5. devices
6. alerts
7. settings
8. command_history

<Expandable title="Migration strategy (preserves all data)">

## Migration Best Practices

### Running Migrations

<Note>
Flyway automatically executes migrations on application startup. Manual execution is rarely needed.
</Note>

```bash
# Migrations run automatically on startup
./gradlew bootRun

# Or in Docker
docker-compose up -d api

# Check migration status
./gradlew flywayInfo

# Validate checksums
./gradlew flywayValidate
```

### Migration Safety

<Warning>
**Never modify executed migrations!** Changing a migration file after execution will cause checksum mismatch and prevent startup.
</Warning>

<CardGroup cols={2}>
  <Card title="✅ Safe Operations" icon="check">
    - Add new columns (nullable)
    - Create new tables
    - Add indexes
    - Insert seed data with `ON CONFLICT DO NOTHING`
  </Card>
  
  <Card title="❌ Dangerous Operations" icon="xmark">
    - Drop columns with data
    - Change column types (may fail)
    - Drop tables (data loss)
    - Modify existing seed data
  </Card>
</CardGroup>

### Rollback Strategy

<Steps>
  <Step title="Database Backup">
    Always backup before running migrations in production
    
    ```bash
    pg_dump -U admin -d postgres > backup_$(date +%Y%m%d).sql
    ```
  </Step>
  
  <Step title="Test in Staging">
    Run migrations in staging environment first
  </Step>
  
  <Step title="Monitor Logs">
    Watch Flyway output for errors
    
    ```bash
    kubectl logs -f deployment/invernaderos-api-prod | grep Flyway
    ```
  </Step>
  
  <Step title="Rollback if Needed">
    Restore from backup if migration fails
    
    ```bash
    psql -U admin -d postgres < backup_20260223.sql
    ```
  </Step>
</Steps>
